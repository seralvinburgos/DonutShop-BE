# pip as the package manager for python projects

# installed venv
python3 -m pip install virtualenv

# create a venv
python -m venv venv

# activate the venv
source venv/bin/activate
# deactivate the venv
deactivate

# installed flask
python3 -m pip install flask


# code


# run the server
python3 server.py

flask --app server --debug run
